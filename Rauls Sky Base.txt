R = game.Workspace.Remote
function PlaceItem(Item, Pos)

end
CurrentItems = {}
function GetAmount()
local Amount = 0
for i, v in pairs(game.Workspace:GetChildren()) do
if CurrentItems[v] ~= true and v:FindFirstChild("IsBuildingMaterial") then
Amount = Amount + 1
end
end
return Amount
end
function PlaceBuilding(Tab, Spot)

for i, v in pairs(game.Workspace:GetChildren()) do
CurrentItems[v] = true
end
local WC = #game.Workspace:GetChildren()

MaxTimes = 500

PlaceItem(Tab[i][1], Tab[i][2]+Spot)
Times = Times + 1
if Times >= MaxTimes then
Times = 0
wait(1)
end
end
repeat
wait()
until GetAmount() >= #Tab
wait(2)
if Val == true then
local TestTimes = 0
